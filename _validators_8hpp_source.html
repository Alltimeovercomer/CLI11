294  typename IsMemberType<typename element_value_type<T>::type>::type)>()) {}

298  template <typename T, typename F> explicit IsMember(T set, F filter_function) {
299  // Get the type of the contained item - requires a container have ::value_type
300  using item_t = typename element_value_type<T>::type;
301  using local_item_t = typename IsMemberType<item_t>::type;
302 
303  // Make a local copy of the filter function, using a std::function if not one already
304  std::function<local_item_t(local_item_t)> filter_fn = filter_function;
305 
306  // This is the type name for help, it will take the current version of the set contents
307  tname_function = [set]() {
308  std::stringstream out;
309  out << detail::type_name<item_t>() << " in {" << detail::join(detail::smart_deref(set), ",") << "}";
310  return out.str();
311  };
312 
313  // This is the function that validates
314  // It stores a copy of the set pointer-like, so shared_ptr will stay alive
315  func = [set, filter_fn](std::string &input) {
316  for(const item_t &v : detail::smart_deref(set)) {
317  local_item_t a = v;
318  local_item_t b;
319  if(!detail::lexical_cast(input, b))
320  throw ValidationError(input); // name is added later
321 
322  // The filter function might be empty, so don't filter if it is.
323  if(filter_fn) {
324  a = filter_fn(a);
325  b = filter_fn(b);
326  }
327 
328  if(a == b) {
329  // Make sure the version in the input string is identical to the one in the set
330  // Requires std::stringstream << be supported on T.
331  if(filter_fn) {
332  std::stringstream out;
333  out << v;
334  input = out.str();
335  }
336 
337  // Return empty error string (success)
338  return std::string();
339  }
340  }
341 
342  // If you reach this point, the result was not found
343  return input + " not in {" + detail::join(detail::smart_deref(set), ",") + "}";
344  };
345  }

366 inline std::pair<std::string, std::string> split_program_name(std::string commandline) {
367  // try to determine the programName
368  std::pair<std::string, std::string> vals;
369  trim(commandline);
370  auto esp = commandline.find_first_of(' ', 1);
371  while(!ExistingFile(commandline.substr(0, esp)).empty()) {
372  esp = commandline.find_first_of(' ', esp + 1);
373  if(esp == std::string::npos) {
374  // if we have reached the end and haven't found a valid file just assume the first argument is the
375  // program name
376  esp = commandline.find_first_of(' ', 1);
377  break;
378  }
379  }
380  vals.first = commandline.substr(0, esp);
381  rtrim(vals.first);
382  // strip the program name
383  vals.second = (esp != std::string::npos) ? commandline.substr(esp + 1) : std::string{};
384  ltrim(vals.second);
385  return vals;
386 }
