558  std::string get_name(bool positional = false, //<[input] Show the positional name
559  bool all_options = false //<[input] Show every option
560  ) const {
561 
562  if(all_options) {
563 
564  std::vector<std::string> name_list;
565 
567  if((positional && pname_.length()) || (snames_.empty() && lnames_.empty()))
568  name_list.push_back(pname_);
569  if((get_items_expected() == 0) && (!fnames_.empty())) {
570  for(const std::string &sname : snames_) {
571  name_list.push_back("-" + sname);
572  if(check_fname(sname)) {
573  name_list.back() += "{false}";
574  }
575  }
576 
577  for(const std::string &lname : lnames_) {
578  name_list.push_back("--" + lname);
579  if(check_fname(lname)) {
580  name_list.back() += "{false}";
581  }
582  }
583  } else {
584  for(const std::string &sname : snames_)
585  name_list.push_back("-" + sname);
586 
587  for(const std::string &lname : lnames_)
588  name_list.push_back("--" + lname);
589  }
590 
591  return detail::join(name_list);
592 
593  } else {
594 
595  // This returns the positional name no matter what
596  if(positional)
597  return pname_;
598 
599  // Prefer long name
600  else if(!lnames_.empty())
601  return std::string("--") + lnames_[0];
602 
603  // Or short name if no long name
604  else if(!snames_.empty())
605  return std::string("-") + snames_[0];
606 
607  // If positional is the only name, it's okay to use that
608  else
609  return pname_;
610  }
611  }

618  void run_callback() {
619 
620  callback_run_ = true;
621 
622  // Run the validators (can change the string)
623  if(!validators_.empty()) {
624  for(std::string &result : results_)
625  for(const std::function<std::string(std::string &)> &vali : validators_) {
626  std::string err_msg;
627 
628  try {
629  err_msg = vali(result);
630  } catch(const ValidationError &err) {
631  throw ValidationError(err.what(), get_name());
632  }
633 
634  if(!err_msg.empty())
635  throw ValidationError(get_name(), err_msg);
636  }
637  }
638 
639  bool local_result;
640 
641  // Num items expected or length of vector, always at least 1
642  // Only valid for a trimming policy
643  int trim_size =
644  std::min<int>(std::max<int>(std::abs(get_items_expected()), 1), static_cast<int>(results_.size()));
645 
646  // Operation depends on the policy setting
647  if(multi_option_policy_ == MultiOptionPolicy::TakeLast) {
648  // Allow multi-option sizes (including 0)
649  results_t partial_result{results_.end() - trim_size, results_.end()};
650  local_result = !callback_(partial_result);
651 
652  } else if(multi_option_policy_ == MultiOptionPolicy::TakeFirst) {
653  results_t partial_result{results_.begin(), results_.begin() + trim_size};
654  local_result = !callback_(partial_result);
655 
656  } else if(multi_option_policy_ == MultiOptionPolicy::Join) {
657  results_t partial_result = {detail::join(results_, "\n")};
658  local_result = !callback_(partial_result);
659 
660  } else {
661  // Exact number required
662  if(get_items_expected() > 0) {
663  if(results_.size() != static_cast<size_t>(get_items_expected()))
664  throw ArgumentMismatch(get_name(), get_items_expected(), results_.size());
665  // Variable length list
666  } else if(get_items_expected() < 0) {
667  // Require that this be a multiple of expected size and at least as many as expected
668  if(results_.size() < static_cast<size_t>(-get_items_expected()) ||
669  results_.size() % static_cast<size_t>(std::abs(get_type_size())) != 0)
670  throw ArgumentMismatch(get_name(), get_items_expected(), results_.size());
671  }
672  local_result = !callback_(results_);
673  }
674 
675  if(local_result)
676  throw ConversionError(get_name(), results_);
677  }

680  bool operator==(const Option &other) const {
681  for(const std::string &sname : snames_)
682  if(other.check_sname(sname))
683  return true;
684  for(const std::string &lname : lnames_)
685  if(other.check_lname(lname))

688  for(const std::string &sname : other.snames_)
689  if(check_sname(sname))
690  return true;
691  for(const std::string &lname : other.lnames_)
692  if(check_lname(lname))
693  return true;
694  return false;
695  }

698  bool check_name(std::string name) const {
699 
700  if(name.length() > 2 && name[0] == '-' && name[1] == '-')
701  return check_lname(name.substr(2));
702  else if(name.length() > 1 && name.front() == '-')
703  return check_sname(name.substr(1));
704  else {
705  std::string local_pname = pname_;
706  if(ignore_underscore_) {
707  local_pname = detail::remove_underscore(local_pname);
708  name = detail::remove_underscore(name);
709  }
710  if(ignore_case_) {
711  local_pname = detail::to_lower(local_pname);
712  name = detail::to_lower(name);
713  }
714  return name == local_pname;
715  }
716  }
