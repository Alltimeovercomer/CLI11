359  Option myopt{option_name, option_description, option_callback, defaulted, this};

369  } else
370  throw OptionAlreadyAdded(myopt.get_name());

1879  void _parse_arg(std::vector<std::string> &args, detail::Classifier current_type) {
1880 
1881  std::string current = args.back();
1882 
1883  std::string arg_name;
1884  std::string value;
1885  std::string rest;
1886 
1887  switch(current_type) {
1888  case detail::Classifier::LONG:
1889  if(!detail::split_long(current, arg_name, value))
1890  throw HorribleError("Long parsed but missing (you should not see this):" + args.back());
1891  break;
1892  case detail::Classifier::SHORT:
1893  if(!detail::split_short(current, arg_name, rest))
1894  throw HorribleError("Short parsed but missing! You should not see this");
1895  break;
1896  case detail::Classifier::WINDOWS:
1897  if(!detail::split_windows(current, arg_name, value))
1898  throw HorribleError("windows option parsed but missing! You should not see this");
1899  break;
1900  default:
1901  throw HorribleError("parsing got called with invalid option! You should not see this");
1902  }
1903 
1904  auto op_ptr =
1905  std::find_if(std::begin(options_), std::end(options_), [arg_name, current_type](const Option_p &opt) {
1906  if(current_type == detail::Classifier::LONG)
1907  return opt->check_lname(arg_name);
1908  if(current_type == detail::Classifier::SHORT)
1909  return opt->check_sname(arg_name);
1910  // this will only get called for detail::Classifier::WINDOWS
1911  return opt->check_lname(arg_name) || opt->check_sname(arg_name);
1912  });
1913 
1914  // Option not found
1915  if(op_ptr == std::end(options_)) {
1916  for(auto &subc : subcommands_) {
1917  if(subc->name_.empty()) {
1918  subc->_parse_arg(args, current_type);
1919  if(subc->missing_.empty()) { // check if it was used and is not in the missing category
1920  return;
1921  } else {
1922  args.push_back(std::move(subc->missing_.front().second));
1923  subc->missing_.clear();
1924  }
1925  }
1926  }
1927  // If a subcommand, try the master command
1928  if(parent_ != nullptr && fallthrough_)
1929  return parent_->_parse_arg(args, current_type);
1930  // Otherwise, add to missing
1931  else {
1932  args.pop_back();
1933  missing_.emplace_back(current_type, current);
1934  return;
1935  }
1936  }
1937 
1938  args.pop_back();
1939 
1940  // Get a reference to the pointer to make syntax bearable
1941  Option_p &op = *op_ptr;
1942 
1943  int num = op->get_items_expected();
1944 
1945  // Make sure we always eat the minimum for unlimited vectors
1946  int collected = 0;
1947  // deal with flag like things
1948  if(num == 0) {
1949  try {
1950  auto res = (value.empty()) ? std ::string("1") : detail::to_flag_value(value);
1951  if(op->check_fname(arg_name)) {
1952  res = (res == "1") ? "-1" : ((res[0] == '-') ? res.substr(1) : std::string("-" + res));
1953  }
1954  op->add_result(res);
1955  parse_order_.push_back(op.get());
1956  } catch(const std::invalid_argument &) {
1957  throw ConversionError::TrueFalse(arg_name);
1958  }
1959  }
1960  // --this=value
1961  else if(!value.empty()) {
1962  // If exact number expected
1963  if(num > 0)
1964  num--;
1965  op->add_result(value);
1966  parse_order_.push_back(op.get());
1967  collected += 1;
1968  // -Trest
1969  } else if(!rest.empty()) {
1970  if(num > 0)
1971  num--;
1972  op->add_result(rest);
1973  parse_order_.push_back(op.get());
1974  rest = "";
1975  collected += 1;
1976  }
1977 
1978  // Unlimited vector parser
1979  if(num < 0) {
1980  while(!args.empty() && _recognize(args.back()) == detail::Classifier::NONE) {
1981  if(collected >= -num) {
1982  // We could break here for allow extras, but we don't
1983 
1984  // If any positionals remain, don't keep eating
1985  if(_count_remaining_positionals() > 0)
1986  break;
1987  }
1988  op->add_result(args.back());
1989  parse_order_.push_back(op.get());
1990  args.pop_back();
1991  collected++;
1992  }
1993 
1994  // Allow -- to end an unlimited list and "eat" it
1995  if(!args.empty() && _recognize(args.back()) == detail::Classifier::POSITIONAL_MARK)
1996  args.pop_back();
1997 
1998  } else {
1999  while(num > 0 && !args.empty()) {
2000  num--;
2001  std::string current_ = args.back();
2002  args.pop_back();
2003  op->add_result(current_);
2004  parse_order_.push_back(op.get());
2005  }
2006 
2007  if(num > 0) {
2008  throw ArgumentMismatch::TypedAtLeast(op->get_name(), num, op->get_type_name());
2009  }
2010  }
2011 
2012  if(!rest.empty()) {
2013  rest = "-" + rest;
2014  args.push_back(rest);
2015  }
2016  }
